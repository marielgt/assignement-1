# Read the data file
data <- read.csv( file=file.choose() )

############################
### Cleaning up the data ###
############################

# Express VMA as an ordinal variable (this is not actually used in the analysis, though )
data$VMA <- ordered( data$VMA, levels=c("60-63", "64-67", "68-79", "80-91", "92-103", "104-115", "116-127", "128-144") )

# VMA as a numeric variable: specifically, this calculates the midpoint of each VMA range
data$VMAn <-    (
                        unlist( lapply( as.character(data$VMA), function(x){ as.numeric(strsplit(x,"-")[[1]][2]) } ) )
                        + unlist( lapply( as.character(data$VMA), function(x){ as.numeric(strsplit(x,"-")[[1]][1]) } ) )
                ) / 2


# Calculate the mean age and SD of age, use them to sphere (z-score) the Age variable, and hold onto them
#	(useful for translating the model coefficients back into original units)
mAge <- mean(data$Age_Month, na.rm=T)
sAge <- sd( data$Age_Month, na.rm=T)
data$zAge <- ( data$Age_Month - mAge ) / sAge


# Calculate the mean age and SD of VMA, use them to sphere (z-score) the VMA variable, and hold onto them
#	(useful for translating the model coefficients back into original units)
mVMAn <- mean(data$VMAn, na.rm=T)
sVMAn <- sd( data$VMAn, na.rm=T)
data$zVMAn <- ( data$VMAn - mVMAn ) / sVMAn


# Calculate the mean age and SD of Raven, use them to sphere (z-score) the Raven variable, and hold onto them
#	(useful for translating the model coefficients back into original units)
mRaven <- mean(data$Raven, na.rm=T)
sRaven <- sd( data$Raven, na.rm=T)
data$zRaven <- ( data$Raven - mRaven ) / sRaven


# Deviation code the Presupposition Type variable. This kind of hard-coding is useful for constructing a model
#	without random correlation parameters (see https://rpubs.com/Reinhold/22193)
data$Existential <- ifelse( data$Types=="Existential", .5, ifelse( data$Types=="Structural", -.5, 0 ) )
data$Factive <- ifelse( data$Types=="Factive", .5, ifelse( data$Types=="Structural", -.5, 0 ) )
data$Lexical <- ifelse( data$Types=="Lexical", .5, ifelse( data$Types=="Structural", -.5, 0 ) )

# Deviation code the Group variable
data$ASD <- ifelse( data$Group=="ASD subjects", .5, -.5 )

### Done cleaning up and coding the data ###
############################################

# load lme4 (you may need to install it first)
library(lme4)

# The biggest model we are interested in: presupposition type, group, VMA, Age, Raven and all 2-way interactions with presupposition type
# The random effects include maximal random effects by Subjects (for the relevant predictors--just the within-subject ones); for items,
#	there are random slopes for the relevant predictors but no intercept and no correlation parameters, to help with convergence.
# Optimizer is set to 'bobyqa' to help avoid convergence errors
system.time( m <- glmer( Correct ~ (Existential+Factive+Lexical)*(ASD+zVMAn+zAge+zRaven) + ((Existential+Factive+Lexical)|Subject) + (0+ASD+zAge+zVMAn+zRaven||Stimuli), data, family="binomial", control=glmerControl(optimizer="bobyqa", optCtrl=list(maxfun=15000)) ) )

# An inverse logit function (for transforming the fixed-effect coefficients into proportions correct)
inverselogit <- function(u){
        return( exp(u) / (1 + exp(u)) )
}
